\iospec@spec{

    \iospec@inputln{
        \iospec@item{N}{\iospec@int}
        \iospec@item{M}{\iospec@int}
    }
    \iospec@assume{2 \le N < 100\;000}
    \iospec@assume{0 \le M < 500\;000}
    \iospec@inputln{
        \iospec@for{u}{N}{
            \iospec@item{W[u]}{\iospec@int}
            \iospec@assume{0 \le W[u] < 1\;000\;000\;000}
        }
    }
    \iospec@for{i}{M}{
        \iospec@inputln{
            \iospec@item{A[i]}{\iospec@int}
            \iospec@item{B[i]}{\iospec@int}
        }
        \iospec@assume{0 \le A[i] < N}
        \iospec@assume{0 \le B[i] < N}
    }
    \iospec@outputln{
        \iospec@item{S}{\iospec@int}
    }
    \iospec@outputln{
        \iospec@for{u}{N}{
            \iospec@item{X[u]}{\iospec@int}
        }
    }
}
